47/* The '__LITERAL' macro generates character literals
48 based on the provided 'type' and 's'.
49 It handles different character types such as
50 'char', 'wchar_t', 'char16_t', 'char32_t', and 'char8_t' (if supported),
51 allowing seamless integration of character literals
52 into template functions with generic character types.
53 Note: '__cpp_char8_t' check ensures compatibility with C++20's 'char8_t'. */
54#ifdef __cpp_char8_t
55#define __LITERAL(type, s) \
56[]() { \
57 if constexpr ( std::is_same_v<type, char> ) { \
58 return s; \
59 } \
60 else if constexpr ( std::is_same_v<type, wchar_t> ) { \
61 return L ## s; \
62 } \
63 else if constexpr ( std::is_same_v<type, char8_t> ) { \
64 return u8 ## s; \
65 } \
66 else if constexpr ( std::is_same_v<type, char16_t> ) { \
67 return u ## s; \
68 } \
69 else if constexpr ( std::is_same_v<type, char32_t> ) { \
70 return U ## s; \
71 } \
72 else { \
73 return s; \
74 } \
75}()
76#else
77#define __LITERAL(type, s) \
78[]() { \

330template <class StringView>
331auto is_single_dashed(StringView word) noexcept {
332 using char_type = typename StringView::value_type;
333
334 return std::size(word) > std::size_t(1)
335 && word[0] == dash<char_type>()
336 && word[1] != dash<char_type>();
337}

347template <class StringView>
348auto is_key(StringView word) noexcept {
349 using char_type = typename StringView::value_type;
350
351 if ( std::size(word) == std::size_t(1) ) {
352 return false;
353 }
354
355 if ( std::size(word) > std::size_t(2) ) {
356 return word[0] == dash<char_type>()
357 && word[2] != dash<char_type>();
358 }
359
360 return word[0] == dash<char_type>()
361 && word[1] != dash<char_type>();
362}

372template <class StringView>
373auto is_complex_boolean_param(StringView word) noexcept {
374 return is_single_dashed(word) && std::size(word) > 2;
375}

401template <class StringView>
402auto split_words(StringView s) {
403 using char_type = typename StringView::value_type;
404
405 auto ret = std::vector<StringView>{};
406
407 auto is_double_quoted = false;
408 auto is_single_quoted = false;
409 auto has_escaped = false;
410 auto has_valid_char = false;
411
412 auto it_first = std::begin(s);
413 auto it_last = std::end(s);
414 auto it_out = std::back_inserter(ret);
415
416 auto pred = std::equal_to<char_type>();
417 auto delim = stream_delim<char_type>();
418 auto is_delim = [&delim](auto ch) {
419 return ch == delim;
420 };
421 auto new_delim = [](auto ch) {
422 if ( ch == single_quote<char_type>() ) {
423 return single_quote<char_type>();
424 }
425 else if ( ch == double_quote<char_type>() ) {
426 return double_quote<char_type>();
427 }
428 else {
429 return stream_delim<char_type>();
430 }
431 };
432
433
434 for (; it_first != it_last; ++it_first) {
435 // jump escape sequence
436 if (has_escaped) {
437 has_escaped = false;
438 continue;
439 }
440
441 // check escape sequence
442 if (
443 has_escaped = pred(
444 *it_first, char_escape<char_type>()
445 )
446 ) {
447 continue;
448 }
449
450 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
451 if (it_first == it_last) {
452 break;
453 }
454 delim = new_delim(*it_first);
455
456 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
457 if (it_first == it_last) {
458 break;
459 }
460
461 auto it_split_last = std::ranges::find_if(
462 it_first, it_last, is_delim
463 );
464 it_out = StringView(it_first, it_split_last);
465 if (it_split_last == it_last) {
466 break;
467 }
468 delim = new_delim(*it_first);
469 it_first = it_split_last;
470 }
471
472 return ret;
473}

987 template <class IntType, class StrArrType>
988 auto parse(IntType argc, StrArrType argv)
989 -> result_tuple_type {
990 auto flattend = string_type();
991
992 for (auto i = decltype(argc)(0); i < argc; ++i) {
993 flattend += string_view_type(argv[i]);
994 flattend += detail::stream_delim<char_type>();
995 }
996
997 return parse( string_view_type(flattend) );
998 }
999
1007 auto parse(string_view_type command_line)
1008 -> result_tuple_type {
1009 result_tuple_type ret;
1010
1011 if (error().has_value()) {
1012 return ret;
1013 }
1014
1015 auto words = detail::split_words(command_line);
1016 if ( words.front() != identifier_ ) {
1017 log_error_invalid_identifier( words.front() );
1018 return ret;
1019 }
1020
1021 auto it_first = std::next(std::begin(words));
1022 auto it_last = std::end(words);
1023
1024 // check first word is key,
1025 if (!detail::is_key(*it_first)) {
1026 log_error_key_not_given();
1027 return ret;
1028 }
1029
1030 while (it_first != it_last) {
1031 if (detail::is_complex_boolean_param(*it_first)) {
1032 if (!parse_complex_keys(*it_first)) {
1033 log_error_wrong_complex_key();
1034 return ret;
1035 }
1036 ++it_first;
1037 continue;
1038 }
1039
1040 auto param_idx = find_param_index(*it_first);
1041 if (!param_idx.has_value()) {
1042 log_error_undefined_key(*it_first);
1043 return ret;
1044 }
1045
1046 auto it_next_key = std::ranges::find_if(
1047 std::next(it_first), it_last,
1048 detail::is_key<string_view_type>
1049 );
1050
1051 // assign (key, next key) words to param
1052 std::ranges::for_each(
1053 std::next(it_first),
1054 it_next_key,
1055 [&, this](auto word) {
1056 arg_stream_ << word << detail::stream_delim<char_type>();
1057 }
1058 );
1059 if ( !assign_arg_by_idx( param_idx.value() ) ) {
1060 log_error_incompatible_argument(*it_first);
1061 return ret;
1062 }
1063
1064 if ( has_unparsed_arguments() ) {
1065 log_error_unparsed_arguments();
1066 return ret;
1067 }
1068
1069 it_first = it_next_key;
1070 } // while (it_first != last)
1071
1072 detail::tuple_conv(params_, ret, [this](const auto& p) {
1073 using none_qualified_t = std::remove_cvref_t<decltype(p)>;
1074 using value_type = typename none_qualified_t::value_type;
1075
1076 if (p.has_value()) {
1077 return p.value();
1078 }
1079
1080 if constexpr ( std::is_base_of_v<
1081 basic_required<value_type, char_type, traits_type>,
1082 none_qualified_t
1083 >
1084 ) {
1085 log_error_required_key_not_given();
1086 }
1087
1088 return value_type{};
1089 });
1090
1091 // TODO: check every required arguments are passed and all constraints are met
1092
1093 return ret;
1094 }

1104 auto parse_complex_keys(string_view_type word) {
1105 /*
1106 fail means at least one of the keys in complex param received
1107 isn't defined as boolean param.
1108 */
1109 auto fail = false;
1110
1111 detail::tuple_for_each(
1112 params_,
1113 [ &fail, keys = detail::remove_dash(word) ](auto& p) mutable {
1114 for (auto key : keys) {
1115 if (p.contains(key)) {
1116 p.assign(true);
1117 if (p.fail()) {
1118 fail = true;
1119 }
1120 }
1121 else {
1122 fail = true;
1123 }
1124 }
1125 }
1126 );
1127
1128 return !fail;
1129 }

1138 auto find_param_index(string_view_type word) {
1139 std::optional<index_type> idx_found{};
1140
1141 detail::tuple_for_each(
1142 params_,
1143 [ &idx_found, this, key = word, i = 0 ](
1144 const auto& p) mutable {
1145 auto dash_removed_key = detail::remove_dash(key);
1146 if ( detail::is_single_dashed(key) ) {
1147 // short param
1148 if ( p.contains(*std::begin(dash_removed_key)) ) {
1149 idx_found = i;
1150 }
1151 }
1152 else if ( p.contains(dash_removed_key) ) {
1153 // long param
1154 idx_found = i;
1155 }
1156 ++i;
1157 }
1158 );
1159
1160 return idx_found;
1161 }
1162
1170 auto assign_arg_by_idx(index_type param_idx) {
1171 bool success = false;
1172
1173 detail::tuple_for_each(
1174 params_,
1175 [ &success, this, param_idx, i = 0 ](auto& p) mutable {
1176 if (param_idx == i) {
1177 this->arg_stream_ >> p;
1178 success = !p.fail();
1179 }
1180 ++i;
1181 }
1182 );
1183
1184 return success;
1185 }
