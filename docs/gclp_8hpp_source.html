48/* The '__LITERAL' macro generates character literals
49 based on the provided 'type' and 's'.
50 It handles different character types such as
51 'char', 'wchar_t', 'char16_t', 'char32_t', and 'char8_t' (if supported),
52 allowing seamless integration of character literals
53 into template functions with generic character types.
54 Note: '__cpp_char8_t' check ensures compatibility with C++20's 'char8_t'. */
55#ifdef __cpp_char8_t
56#define __LITERAL(type, s) \
57[]() { \
58 if constexpr ( std::is_same_v<type, char> ) { \
59 return s; \
60 } \
61 else if constexpr ( std::is_same_v<type, wchar_t> ) { \
62 return L ## s; \
63 } \
64 else if constexpr ( std::is_same_v<type, char8_t> ) { \
65 return u8 ## s; \
66 } \
67 else if constexpr ( std::is_same_v<type, char16_t> ) { \
68 return u ## s; \
69 } \
70 else if constexpr ( std::is_same_v<type, char32_t> ) { \
71 return U ## s; \
72 } \
73 else { \
74 return s; \
75 } \
76}()
77#else

629template <class StringView>
630bool is_single_dashed(StringView word) noexcept {
631 using char_type = typename StringView::value_type;
632
633 return std::size(word) > std::size_t(1)
634 && word[0] == dash<char_type>()
635 && word[1] != dash<char_type>();
636}

646template <class StringView>
647bool is_key(StringView word) noexcept {
648 using char_type = typename StringView::value_type;
649
650 if ( std::size(word) == std::size_t(1) ) {
651 return false;
652 }
653
654 if ( std::size(word) > std::size_t(2) ) {
655 return word[0] == dash<char_type>()
656 && word[2] != dash<char_type>();
657 }
658
659 return word[0] == dash<char_type>()
660 && word[1] != dash<char_type>();
661}
662
671template <class StringView>
672bool is_complex_boolean_param(StringView word) noexcept {
673 return is_single_dashed(word) && std::size(word) > 2;
674}

700template <class StringView>
701std::vector<StringView> split_words(StringView s) {
702 using char_type = typename StringView::value_type;
703
704 auto ret = std::vector<StringView>{};
705
706 auto is_double_quoted = false;
707 auto is_single_quoted = false;
708 auto has_escaped = false;
709 auto has_valid_char = false;
710
711 auto it_first = std::begin(s);
712 auto it_last = std::end(s);
713 auto it_out = std::back_inserter(ret);
714
715 auto pred = std::equal_to<char_type>();
716 auto delim = stream_delim<char_type>();
717 auto is_delim = [&delim](auto ch) {
718 return ch == delim;
719 };
720 auto new_delim = [](auto ch) {
721 if ( ch == single_quote<char_type>() ) {
722 return single_quote<char_type>();
723 }
724 else if ( ch == double_quote<char_type>() ) {
725 return double_quote<char_type>();
726 }
727 else {
728 return stream_delim<char_type>();
729 }
730 };
731
732
733 for (; it_first != it_last; ++it_first) {
734 // jump escape sequence
735 if (has_escaped) {
736 has_escaped = false;
737 continue;
738 }
739
740 // check escape sequence
741 if (
742 has_escaped = pred(
743 *it_first, char_escape<char_type>()
744 )
745 ) {
746 continue;
747 }
748
749 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
750 if (it_first == it_last) {
751 break;
752 }
753 delim = new_delim(*it_first);
754
755 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
756 if (it_first == it_last) {
757 break;
758 }
759
760 auto it_split_last = std::ranges::find_if(
761 it_first, it_last, is_delim
762 );
763 it_out = StringView(it_first, it_split_last);
764 if (it_split_last == it_last) {
765 break;
766 }
767 delim = new_delim(*it_first);
768 it_first = it_split_last;
769 }
770
771 return ret;
772}

1513 result_tuple_type parse(string_view_type command_line) {
1514 // Check if parameter has been already assigned a value
1515 // to guarantee that the parameters will have been assigned a value
1516 // only once.
1517 auto has_duplicated_assignments = [b_assigned
1518 = std::bitset< std::tuple_size_v<result_tuple_type> >(0u),
1519 this
1520 ](auto word) mutable {
1521 bool ret = false;
1522
1523 auto map_with_key = [&](auto key) {
1524 detail::tuple_for_each(
1525 this->params_,
1526 [&, i = 0](auto& p) mutable {
1527 if ( p.contains(key) ) {
1528 if (b_assigned.test(i)) {
1529 ret = true;
1530 }
1531 b_assigned.set(i);
1532 }
1533 ++i;
1534 }
1535 );
1536 };
1537
1538 if ( detail::is_complex_boolean_param(word) ) {
1539 for (auto key : detail::remove_dash(word)) {
1540 map_with_key(key);
1541 }
1542 }
1543 else {
1544 auto key = detail::remove_dash(word);
1545 if ( std::size(key) > 1) {
1546 map_with_key(key);
1547 }
1548 else { // short option must be treated as a character
1549 // rather than string.
1550 map_with_key( *std::begin(key) );
1551 }
1552 }
1553
1554 return ret;
1555 };
1556
1557 if (error().has_value()) {
1558 return {};
1559 }
1560
1561 auto words = detail::split_words(command_line);
1562
1563 if ( std::empty(words) ) {
1564 log_error_identifier_not_given();
1565 return {};
1566 }
1567
1568 if ( words.front() != identifier_ ) {
1569 log_error_invalid_identifier( words.front() );
1570 return {};
1571 }
1572
1573 auto it_first = std::next(std::begin(words));
1574 auto it_last = std::end(words);
1575
1576 if (it_first == it_last) {
1577 // there's no argument to parse.
1578 return make_parsed_result();
1579 }
1580
1581 // check first word is key,
1582 if (!detail::is_key(*it_first)) {
1583 log_error_key_not_given();
1584 return {};
1585 }
1586
1587 while (it_first != it_last) {
1588 if ( detail::is_complex_boolean_param(*it_first) ) {
1589 if ( !parse_complex_keys(*it_first) ) {
1590 log_error_wrong_complex_key(*it_first);
1591 return {};
1592 }
1593 if ( has_duplicated_assignments(*it_first) ) {
1594 log_error_duplicated_assignments(*it_first);
1595 }
1596 ++it_first;
1597 continue;
1598 }
1599
1600 auto param_idx = find_param_index(*it_first);
1601 if ( !param_idx.has_value() ) {
1602 log_error_undefined_key(*it_first);
1603 return {};
1604 }
1605
1606 auto it_next_key = std::ranges::find_if(
1607 std::next(it_first), it_last,
1608 detail::is_key<string_view_type>
1609 );
1610
1611 // assign (key, next key) words to param
1612 std::ranges::for_each(
1613 std::next(it_first),
1614 it_next_key,
1615 [&, this](auto word) {
1616 arg_stream_ << word << detail::stream_delim<char_type>();
1617 }
1618 );
1619 if ( !assign_arg_by_idx( param_idx.value() ) ) {
1620 log_error_incompatible_argument(*it_first);
1621 return {};
1622 }
1623
1624 if ( has_duplicated_assignments(*it_first) ) {
1625 log_error_duplicated_assignments(*it_first);
1626 return {};
1627 }
1628
1629 if ( has_unparsed_arguments() ) {
1630 log_error_unparsed_arguments();
1631 return {};
1632 }
1633
1634 it_first = it_next_key;
1635 } // while (it_first != last)
1636
1637 // TODO: check every required arguments are passed and all constraints are met
1638
1639 return make_parsed_result();
1640 }

1700 bool parse_complex_keys(string_view_type word) {
1701 /*
1702 fail means at least one of the keys in complex param received
1703 isn't defined as boolean param.
1704 */
1705 auto fail = false;
1706
1707 for (auto key : detail::remove_dash(word)) {
1708 auto key_found = false;
1709
1710 detail::tuple_for_each(
1711 params_,
1712 [ &fail, &key_found, key ](auto& p) {
1713 if (p.contains(key)) {
1714 key_found = true;
1715 p.assign(true);
1716 if (p.fail()) {
1717 fail = true;
1718 }
1719 }
1720 }
1721 );
1722
1723 if (!key_found || fail) {
1724 return false;
1725 }
1726 }
1727
1728 return true;
1729 }

1738 std::optional<index_type> find_param_index(string_view_type word) {
1739 auto idx_found = std::optional<index_type>();
1740
1741 detail::tuple_for_each(
1742 params_,
1743 [ &idx_found, this, key = word, i = 0 ](
1744 const auto& p) mutable {
1745 auto dash_removed_key = detail::remove_dash(key);
1746 if ( detail::is_single_dashed(key) ) {
1747 // short param
1748 if ( p.contains(*std::begin(dash_removed_key)) ) {
1749 idx_found = i;
1750 }
1751 }
1752 else if ( p.contains(dash_removed_key) ) {
1753 // long param
1754 idx_found = i;
1755 }
1756 ++i;
1757 }
1758 );
1759
1760 return idx_found;
1761 }

1770 bool assign_arg_by_idx(index_type param_idx) {
1771 bool success = false;
1772
1773 detail::tuple_for_each(
1774 params_,
1775 [ &success, this, param_idx, i = 0 ](auto& p) mutable {
1776 using value_type = typename std::remove_cvref_t<decltype(p)>
1777 ::value_type;
1778
1779 if (param_idx == i) {
1780 // when extracting string_type,
1781 // all characters in the stream regardless of white space should be extracted.
1782 if constexpr (std::is_base_of_v<string_type, value_type>) {
1783 auto acc = string_type();
1784 for (auto&& s : std::views::istream<string_type>(this->arg_stream_)) {
1785 acc += s;
1786 acc += detail::stream_delim<char_type>();
1787 }
1788 acc.pop_back();
1789 p.assign( std::move(acc) );
1790 this->arg_stream_.clear();
1791 }
1792 // when extracting boolean type,
1793 // omitting argument means the value is true
1794 else if constexpr (std::is_base_of_v<bool, value_type>
1795 || std::is_same_v<bool, value_type>
1796 ) {
1797 if (!has_unparsed_arguments()) {
1798 p.assign(true);
1799 }
1800 else {
1801 this->arg_stream_ >> p;
1802 }
1803 }
1804 else {
1805 this->arg_stream_ >> p;
1806 }
1807 success = !p.fail();
1808 }
1809 ++i;
1810 }
1811 );
1812
1813 return success;
1814 }
