48/* The '__LITERAL' macro generates character literals
49 based on the provided 'type' and 's'.
50 It handles different character types such as
51 'char', 'wchar_t', 'char16_t', 'char32_t', and 'char8_t' (if supported),
52 allowing seamless integration of character literals
53 into template functions with generic character types.
54 Note: '__cpp_char8_t' check ensures compatibility with C++20's 'char8_t'. */
55#ifdef __cpp_char8_t
56#define __LITERAL(type, s) \
57[]() { \
58 if constexpr ( std::is_same_v<type, char> ) { \
59 return s; \
60 } \
61 else if constexpr ( std::is_same_v<type, wchar_t> ) { \
62 return L ## s; \
63 } \
64 else if constexpr ( std::is_same_v<type, char8_t> ) { \
65 return u8 ## s; \
66 } \
67 else if constexpr ( std::is_same_v<type, char16_t> ) { \
68 return u ## s; \
69 } \
70 else if constexpr ( std::is_same_v<type, char32_t> ) { \
71 return U ## s; \
72 } \
73 else { \
74 return s; \
75 } \
76}()
77#else

629template <class StringView>
630bool is_single_dashed(StringView word) noexcept {
631 using char_type = typename StringView::value_type;
632
633 return std::size(word) > std::size_t(1)
634 && word[0] == dash<char_type>()
635 && word[1] != dash<char_type>();
636}

646template <class StringView>
647bool is_key(StringView word) noexcept {
648 using char_type = typename StringView::value_type;
649
650 if ( std::size(word) == std::size_t(1) ) {
651 return false;
652 }
653
654 if ( std::size(word) > std::size_t(2) ) {
655 return word[0] == dash<char_type>()
656 && word[2] != dash<char_type>();
657 }
658
659 return word[0] == dash<char_type>()
660 && word[1] != dash<char_type>();
661}
662
671template <class StringView>
672bool is_complex_boolean_param(StringView word) noexcept {
673 return is_single_dashed(word) && std::size(word) > 2;
674}

700template <class StringView>
701std::vector<StringView> split_words(StringView s) {
702 using char_type = typename StringView::value_type;
703
704 auto ret = std::vector<StringView>{};
705
706 auto is_double_quoted = false;
707 auto is_single_quoted = false;
708 auto has_escaped = false;
709 auto has_valid_char = false;
710
711 auto it_first = std::begin(s);
712 auto it_last = std::end(s);
713 auto it_out = std::back_inserter(ret);
714
715 auto pred = std::equal_to<char_type>();
716 auto delim = stream_delim<char_type>();
717 auto is_delim = [&delim](auto ch) {
718 return ch == delim;
719 };
720 auto new_delim = [](auto ch) {
721 if ( ch == single_quote<char_type>() ) {
722 return single_quote<char_type>();
723 }
724 else if ( ch == double_quote<char_type>() ) {
725 return double_quote<char_type>();
726 }
727 else {
728 return stream_delim<char_type>();
729 }
730 };
731
732
733 for (; it_first != it_last; ++it_first) {
734 // jump escape sequence
735 if (has_escaped) {
736 has_escaped = false;
737 continue;
738 }
739
740 // check escape sequence
741 if (
742 has_escaped = pred(
743 *it_first, char_escape<char_type>()
744 )
745 ) {
746 continue;
747 }
748
749 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
750 if (it_first == it_last) {
751 break;
752 }
753 delim = new_delim(*it_first);
754
755 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
756 if (it_first == it_last) {
757 break;
758 }
759
760 auto it_split_last = std::ranges::find_if(
761 it_first, it_last, is_delim
762 );
763 it_out = StringView(it_first, it_split_last);
764 if (it_split_last == it_last) {
765 break;
766 }
767 delim = new_delim(*it_first);
768 it_first = it_split_last;
769 }
770
771 return ret;
772}

1397 result_tuple_type parse(string_view_type command_line) {
1398 // Check if parameter has been already assigned a value
1399 // to guarantee that the parameters will have been assigned a value
1400 // only once.
1401 auto has_duplicated_assignments = [b_assigned
1402 = std::bitset< std::tuple_size_v<result_tuple_type> >(0u),
1403 this
1404 ](auto word) mutable {
1405 bool ret = false;
1406
1407 auto map_with_key = [&](auto key) {
1408 detail::tuple_for_each(
1409 this->params_,
1410 [&, i = 0](auto& p) mutable {
1411 if ( p.contains(key) ) {
1412 if (b_assigned.test(i)) {
1413 ret = true;
1414 }
1415 b_assigned.set(i);
1416 }
1417 ++i;
1418 }
1419 );
1420 };
1421
1422 if ( detail::is_complex_boolean_param(word) ) {
1423 for (auto key : detail::remove_dash(word)) {
1424 map_with_key(key);
1425 }
1426 }
1427 else {
1428 auto key = detail::remove_dash(word);
1429 if ( std::size(key) > 1) {
1430 map_with_key(key);
1431 }
1432 else { // short option must be treated as a character
1433 // rather than string.
1434 map_with_key( *std::begin(key) );
1435 }
1436 }
1437
1438 return ret;
1439 };
1440
1441 if (error().has_value()) {
1442 return {};
1443 }
1444
1445 auto words = detail::split_words(command_line);
1446
1447 if ( std::empty(words) ) {
1448 log_error_identifier_not_given();
1449 return {};
1450 }
1451
1452 if ( words.front() != identifier_ ) {
1453 log_error_invalid_identifier( words.front() );
1454 return {};
1455 }
1456
1457 auto it_first = std::next(std::begin(words));
1458 auto it_last = std::end(words);
1459
1460 if (it_first == it_last) {
1461 // there's no argument to parse.
1462 return make_parsed_result();
1463 }
1464
1465 // check first word is key,
1466 if (!detail::is_key(*it_first)) {
1467 log_error_key_not_given();
1468 return {};
1469 }
1470
1471 while (it_first != it_last) {
1472 if ( detail::is_complex_boolean_param(*it_first) ) {
1473 if ( !parse_complex_keys(*it_first) ) {
1474 log_error_wrong_complex_key(*it_first);
1475 return {};
1476 }
1477 if ( has_duplicated_assignments(*it_first) ) {
1478 log_error_duplicated_assignments(*it_first);
1479 }
1480 ++it_first;
1481 continue;
1482 }
1483
1484 auto param_idx = find_param_index(*it_first);
1485 if ( !param_idx.has_value() ) {
1486 log_error_undefined_key(*it_first);
1487 return {};
1488 }
1489
1490 auto it_next_key = std::ranges::find_if(
1491 std::next(it_first), it_last,
1492 detail::is_key<string_view_type>
1493 );
1494
1495 // assign (key, next key) words to param
1496 std::ranges::for_each(
1497 std::next(it_first),
1498 it_next_key,
1499 [&, this](auto word) {
1500 arg_stream_ << word << detail::stream_delim<char_type>();
1501 }
1502 );
1503 if ( !assign_arg_by_idx( param_idx.value() ) ) {
1504 log_error_incompatible_argument(*it_first);
1505 return {};
1506 }
1507
1508 if ( has_duplicated_assignments(*it_first) ) {
1509 log_error_duplicated_assignments(*it_first);
1510 return {};
1511 }
1512
1513 if ( has_unparsed_arguments() ) {
1514 log_error_unparsed_arguments();
1515 return {};
1516 }
1517
1518 it_first = it_next_key;
1519 } // while (it_first != last)
1520
1521 // TODO: check every required arguments are passed and all constraints are met
1522
1523 return make_parsed_result();
1524 }

1528 result_tuple_type ret;

1530 detail::tuple_conv(params_, ret, [this](const auto& p) {
1531 using none_qualified_t = std::remove_cvref_t<decltype(p)>;
1532 using value_type = typename none_qualified_t::value_type;
1533
1534 if (p.has_value()) {
1535 return p.value();
1536 }
1537
1538 if constexpr ( std::is_base_of_v<
1539 basic_required<value_type, char_type, traits_type>,
1540 none_qualified_t
1541 >
1542 ) {
1543 log_error_required_key_not_given();
1544 }
1545
1546 return value_type{};
1547 });
1548
1549 return ret;

1566 bool parse_complex_keys(string_view_type word) {
1567 /*
1568 fail means at least one of the keys in complex param received
1569 isn't defined as boolean param.
1570 */
1571 auto fail = false;
1572
1573 for (auto key : detail::remove_dash(word)) {
1574 auto key_found = false;
1575
1576 detail::tuple_for_each(
1577 params_,
1578 [ &fail, &key_found, key ](auto& p) {
1579 if (p.contains(key)) {
1580 key_found = true;
1581 p.assign(true);
1582 if (p.fail()) {
1583 fail = true;
1584 }
1585 }
1586 }
1587 );
1588
1589 if (!key_found || fail) {
1590 return false;
1591 }
1592 }
1593
1594 return true;
1595 }

1604 std::optional<index_type> find_param_index(string_view_type word) {
1605 auto idx_found = std::optional<index_type>();
1606
1607 detail::tuple_for_each(
1608 params_,
1609 [ &idx_found, this, key = word, i = 0 ](
1610 const auto& p) mutable {
1611 auto dash_removed_key = detail::remove_dash(key);
1612 if ( detail::is_single_dashed(key) ) {
1613 // short param
1614 if ( p.contains(*std::begin(dash_removed_key)) ) {
1615 idx_found = i;
1616 }
1617 }
1618 else if ( p.contains(dash_removed_key) ) {
1619 // long param
1620 idx_found = i;
1621 }
1622 ++i;
1623 }
1624 );
1625
1626 return idx_found;

1636 bool assign_arg_by_idx(index_type param_idx) {
1637 bool success = false;
1638
1639 detail::tuple_for_each(
1640 params_,
1641 [ &success, this, param_idx, i = 0 ](auto& p) mutable {
1642 using value_type = typename std::remove_cvref_t<decltype(p)>
1643 ::value_type;
1644
1645 if (param_idx == i) {
1646 // when extracting string_type,
1647 // all characters in the stream regardless of white space should be extracted.
1648 if constexpr (std::is_base_of_v<string_type, value_type>) {
1649 auto acc = string_type();
1650 for (auto&& s : std::views::istream<string_type>(this->arg_stream_)) {
1651 acc += s;
1652 acc += detail::stream_delim<char_type>();
1653 }
1654 acc.pop_back();
1655 p.assign( std::move(acc) );
1656 this->arg_stream_.clear();
1657 }
1658 // when extracting boolean type,
1659 // omitting argument means the value is true
1660 else if constexpr (std::is_base_of_v<bool, value_type>
1661 || std::is_same_v<bool, value_type>
1662 ) {
1663 if (!has_unparsed_arguments()) {
1664 p.assign(true);
1665 }
1666 else {
1667 this->arg_stream_ >> p;
1668 }
1669 }
1670 else {
1671 this->arg_stream_ >> p;
1672 }
1673 success = !p.fail();
1674 }
1675 ++i;
1676 }
1677 );
1678
1679 return success;
1680 }
