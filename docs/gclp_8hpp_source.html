48/* The '__LITERAL' macro generates character literals
49 based on the provided 'type' and 's'.
50 It handles different character types such as
51 'char', 'wchar_t', 'char16_t', 'char32_t', and 'char8_t' (if supported),
52 allowing seamless integration of character literals
53 into template functions with generic character types.
54 Note: '__cpp_char8_t' check ensures compatibility with C++20's 'char8_t'. */

56#define __LITERAL(type, s) \
57[]() { \
58 if constexpr ( std::is_same_v<type, char> ) { \
59 return s; \
60 } \
61 else if constexpr ( std::is_same_v<type, wchar_t> ) { \
62 return L ## s; \
63 } \
64 else if constexpr ( std::is_same_v<type, char8_t> ) { \
65 return u8 ## s; \
66 } \
67 else if constexpr ( std::is_same_v<type, char16_t> ) { \
68 return u ## s; \
69 } \

71 return U ## s; \

76}()

625template <class StringView>
626bool is_single_dashed(StringView word) noexcept {
627 using char_type = typename StringView::value_type;
628
629 return std::size(word) > std::size_t(1)
630 && word[0] == dash<char_type>()
631 && word[1] != dash<char_type>();
632}

642template <class StringView>
643bool is_key(StringView word) noexcept {
644 using char_type = typename StringView::value_type;
645
646 if ( std::size(word) == std::size_t(1) ) {
647 return false;
648 }
649
650 if ( std::size(word) > std::size_t(2) ) {
651 return word[0] == dash<char_type>()
652 && word[2] != dash<char_type>();
653 }
654
655 return word[0] == dash<char_type>()
656 && word[1] != dash<char_type>();
657}

667template <class StringView>
668bool is_complex_boolean_param(StringView word) noexcept {
669 return is_single_dashed(word) && std::size(word) > 2;

696template <class StringView>
697std::vector<StringView> split_words(StringView s) {
698 using char_type = typename StringView::value_type;
699
700 auto ret = std::vector<StringView>{};
701
702 auto is_double_quoted = false;
703 auto is_single_quoted = false;
704 auto has_escaped = false;
705 auto has_valid_char = false;
706
707 auto it_first = std::begin(s);
708 auto it_last = std::end(s);
709 auto it_out = std::back_inserter(ret);
710
711 auto pred = std::equal_to<char_type>();
712 auto delim = stream_delim<char_type>();
713 auto is_delim = [&delim](auto ch) {
714 return ch == delim;
715 };
716 auto new_delim = [](auto ch) {
717 if ( ch == single_quote<char_type>() ) {
718 return single_quote<char_type>();
719 }
720 else if ( ch == double_quote<char_type>() ) {
721 return double_quote<char_type>();
722 }
723 else {
724 return stream_delim<char_type>();
725 }
726 };
727
728
729 for (; it_first != it_last; ++it_first) {
730 // jump escape sequence
731 if (has_escaped) {
732 has_escaped = false;
733 continue;
734 }
735
736 // check escape sequence
737 if (
738 has_escaped = pred(
739 *it_first, char_escape<char_type>()
740 )
741 ) {
742 continue;
743 }
744
745 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
746 if (it_first == it_last) {
747 break;
748 }
749 delim = new_delim(*it_first);
750
751 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
752 if (it_first == it_last) {
753 break;
754 }
755
756 auto it_split_last = std::ranges::find_if(
757 it_first, it_last, is_delim
758 );
759 it_out = StringView(it_first, it_split_last);
760 if (it_split_last == it_last) {
761 break;
762 }
763 delim = new_delim(*it_first);
764 it_first = it_split_last;
765 }
766
767 return ret;
768}

1346 result_tuple_type parse(string_view_type command_line) {
1347 result_tuple_type ret;
1348
1349 // Check if parameter has been already assigned a value
1350 // to guarantee that the parameters will have been assigned a value
1351 // only once.
1352 auto has_duplicated_assignments = [b_assigned
1353 = std::bitset< std::tuple_size_v<result_tuple_type> >(0u),
1354 this
1355 ](auto word) mutable {
1356 bool ret = false;
1357
1358 auto map_with_key = [&](auto key) {
1359 detail::tuple_for_each(
1360 this->params_,
1361 [&, i = 0](auto& p) mutable {
1362 if ( p.contains(key) ) {
1363 if (b_assigned.test(i)) {
1364 ret = true;
1365 }
1366 b_assigned.set(i);
1367 }
1368 ++i;
1369 }
1370 );
1371 };
1372
1373 if ( detail::is_complex_boolean_param(word) ) {
1374 for (auto key : detail::remove_dash(word)) {
1375 map_with_key(key);
1376 }
1377 }
1378 else {
1379 auto key = detail::remove_dash(word);
1380 if ( std::size(key) > 1) {
1381 map_with_key(key);
1382 }
1383 else { // short option must be treated as a character
1384 // rather than string.
1385 map_with_key( *std::begin(key) );
1386 }
1387 }
1388
1389 return ret;
1390 };
1391
1392 if (error().has_value()) {
1393 return ret;
1394 }
1395
1396 auto words = detail::split_words(command_line);
1397 if ( words.front() != identifier_ ) {
1398 log_error_invalid_identifier( words.front() );
1399 return ret;
1400 }
1401
1402 auto it_first = std::next(std::begin(words));
1403 auto it_last = std::end(words);
1404
1405 // check first word is key,
1406 if (!detail::is_key(*it_first)) {
1407 log_error_key_not_given();
1408 return ret;
1409 }
1410
1411 while (it_first != it_last) {
1412 if ( detail::is_complex_boolean_param(*it_first) ) {
1413 if ( !parse_complex_keys(*it_first) ) {
1414 log_error_wrong_complex_key(*it_first);
1415 return ret;
1416 }
1417 if ( has_duplicated_assignments(*it_first) ) {
1418 log_error_duplicated_assignments(*it_first);
1419 }
1420 ++it_first;
1421 continue;
1422 }
1423
1424 auto param_idx = find_param_index(*it_first);
1425 if ( !param_idx.has_value() ) {
1426 log_error_undefined_key(*it_first);
1427 return ret;
1428 }
1429
1430 auto it_next_key = std::ranges::find_if(
1431 std::next(it_first), it_last,
1432 detail::is_key<string_view_type>
1433 );
1434
1435 // assign (key, next key) words to param
1436 std::ranges::for_each(
1437 std::next(it_first),
1438 it_next_key,
1439 [&, this](auto word) {
1440 arg_stream_ << word << detail::stream_delim<char_type>();
1441 }
1442 );
1443 if ( !assign_arg_by_idx( param_idx.value() ) ) {
1444 log_error_incompatible_argument(*it_first);
1445 return ret;
1446 }
1447
1448 if ( has_duplicated_assignments(*it_first) ) {
1449 log_error_duplicated_assignments(*it_first);
1450 }
1451
1452 if ( has_unparsed_arguments() ) {
1453 log_error_unparsed_arguments();
1454 return ret;
1455 }
1456
1457 it_first = it_next_key;
1458 } // while (it_first != last)

1460 detail::tuple_conv(params_, ret, [this](const auto& p) {
1461 using none_qualified_t = std::remove_cvref_t<decltype(p)>;
1462 using value_type = typename none_qualified_t::value_type;
1463
1464 if (p.has_value()) {
1465 return p.value();
1466 }
1467
1468 if constexpr ( std::is_base_of_v<
1469 basic_required<value_type, char_type, traits_type>,
1470 none_qualified_t
1471 >
1472 ) {
1473 log_error_required_key_not_given();
1474 }
1475
1476 return value_type{};
1477 });
1478
1479 // TODO: check every required arguments are passed and all constraints are met
1480
1481 return ret;
1482 }

1601 ++i;
1602 }
