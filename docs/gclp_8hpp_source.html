751 auto is_double_quoted = false;
752 auto is_single_quoted = false;
753 auto has_escaped = false;
754 auto has_valid_char = false;
755
756 auto it_first = std::begin(s);
757 auto it_last = std::end(s);
758 auto it_out = std::back_inserter(ret);
759
760 auto pred = std::equal_to<char_type>();
761 auto delim = stream_delim<char_type>();
762 auto is_delim = [&delim](auto ch) {
763 return ch == delim;
764 };
765 auto new_delim = [](auto ch) {
766 if ( ch == single_quote<char_type>() ) {
767 return single_quote<char_type>();
768 }
769 else if ( ch == double_quote<char_type>() ) {
770 return double_quote<char_type>();
771 }
772 else {
773 return stream_delim<char_type>();
774 }
775 };
776
777
778 for (; it_first != it_last; ++it_first) {
779 // jump escape sequence
780 if (has_escaped) {
781 has_escaped = false;
782 continue;
783 }
784
785 // check escape sequence
786 if (
787 has_escaped = pred(
788 *it_first, char_escape<char_type>()
789 )
790 ) {
791 continue;
792 }
793
794 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
795 if (it_first == it_last) {
796 break;
797 }
798 delim = new_delim(*it_first);
799
800 it_first = std::ranges::find_if_not(it_first, it_last, is_delim);
801 if (it_first == it_last) {
